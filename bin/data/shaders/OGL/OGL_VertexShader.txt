#version 420 core

layout (location = 0) in vec3 aPosition;

layout(std140, binding = 0) uniform ConstantBuffer1
{
    mat4 View;
    mat4 Projection;
};

layout(std140, binding = 1) uniform ConstantBuffer2
{
        mat4 World;
        vec4 vMeshColor;
};

void main()
{
    gl_Position = Projection * View * World * vec4(aPosition.xyz, 1.0);
}
